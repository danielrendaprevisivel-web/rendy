async function checkAdminAccess() {
  const { data: { session } } = await supabaseClient.auth.getSession();
  if (!session) {
    // Redirect to login or show message
    alert('Faça login primeiro');
    return;
  }
  // Check if user is admin (you need to have admin role in your users table)
  const { data: userData } = await supabaseClient
    .from('users')
    .select('role')
    .eq('id', session.user.id)
    .single();
  
  if (!userData || userData.role !== 'admin') {
    alert('Acesso restrito');
    return;
  }
  showAdminPanel();
}

function showAdminPanel() {
  document.getElementById('admin-panel').classList.add('show');
  loadAdminStudents();
}

async function loadAdminStudents() {
  // Query chat_history for all users with summaries
  const { data: histories, error } = await supabaseClient
    .from('chat_history')
    .select('user_id, summary_data, updated_at')
    .not('summary_data', 'is', null)
    .order('updated_at', { ascending: false });

  if (error) {
    console.error('Error loading students:', error);
    return;
  }

  // Get unique user_ids
  const userIds = [...new Set(histories.map(h => h.user_id))];

  // Fetch user emails from auth.users
  const { data: users } = await supabaseClient.auth.admin.listUsers();

  // Build student list
  const studentList = userIds.map(userId => {
    const user = users.users.find(u => u.id === userId);
    const history = histories.find(h => h.user_id === userId);
    return {
      id: userId,
      email: user?.email || 'Unknown',
      nicho: history?.summary_data?.nicho || '—',
      subnicho: history?.summary_data?.subnicho || '—',
      updatedAt: history?.updated_at
    };
  });

  renderStudentList(studentList);
}

function renderStudentList(students) {
  const list = document.getElementById('admin-student-list');
  list.innerHTML = students.map(s => `
    <div class="admin-student-item" onclick="openStudentDetail('${s.id}')">
      <div class="admin-student-email">${s.email}</div>
      <div class="admin-student-meta">${s.nicho} › ${s.subnicho} · ${new Date(s.updatedAt).toLocaleDateString('pt-BR')}</div>
    </div>
  `).join('');
}
